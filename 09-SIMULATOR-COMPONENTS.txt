export function IssueCard({ issue }: { issue: Issue }) {
  const { openIssueDetail } = useSimulatorStore();
  
  return (
    <div
      className="bg-white rounded shadow-atlassian-sm p-3 cursor-pointer hover:shadow-atlassian-md transition-shadow"
      onClick={() => openIssueDetail(issue)}
      data-testid="board-card"
      data-issue-id={issue.id}
    >
      {issue.epic && (
        <EpicBadge epic={issue.epic} className="mb-2" />
      )}
      
      <p className="text-sm text-atlassian-neutral-900 line-clamp-2 mb-3">
        {issue.summary}
      </p>
      
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <IssueTypeIcon type={issue.type} size={16} />
          <span className="text-xs text-atlassian-neutral-600">{issue.key}</span>
          <PriorityIcon priority={issue.priority} size={16} />
          {issue.storyPoints && (
            <span className="text-xs bg-atlassian-neutral-200 px-1.5 py-0.5 rounded">
              {issue.storyPoints}
            </span>
          )}
        </div>
        
        {issue.assignee && (
          <AssigneeAvatar user={issue.assignee} size={24} />
        )}
      </div>
    </div>
  );
}

================================================================================

Main wrapper that assembles all simulator components.

CODE STRUCTURE:
```tsx
export function JiraSimulator({ 
  initialState,
  lessonMode = false 
}: { 
  initialState?: Partial<SimulatorState>;
  lessonMode?: boolean;
}) {
  const { currentView, issueDetailOpen } = useSimulatorStore();
  
  // Initialize state on mount
  useEffect(() => {
    if (initialState) {
      // Apply initial state for lesson
    }
  }, [initialState]);
  
  return (
    <SimulatorProvider>
      <div className="h-screen flex flex-col bg-atlassian-neutral-100">
        {/* Global Navigation */}
        <GlobalNav />
        
        <div className="flex flex-1 pt-14">
          {/* Sidebar */}
          <Sidebar />
          
          {/* Main Content */}
          <main className="flex-1 ml-60">
            {currentView === 'board' && <BoardView />}
            {currentView === 'backlog' && <BacklogView />}
            {currentView === 'timeline' && <TimelineView />}
            {currentView === 'settings' && <ProjectSettings />}
          </main>
        </div>
        
        {/* Issue Detail Panel */}
        {issueDetailOpen && <IssueDetailPanel />}
        
        {/* Create Issue Modal */}
        <CreateIssueModal />
        
        {/* Search Dialog */}
        <SearchDialog />
      </div>
    </SimulatorProvider>
  );
}
```
